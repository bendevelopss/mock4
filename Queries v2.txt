ACTIVITIES SELECTION
SELECT DISTINCT activities.description, activities.station, activities.qty 
FROM activities, dailyacco
WHERE activities.status='ACTIVE'
AND
activities.date=(SELECT date FROM dailyacco WHERE date='?')

DAILYLABOR SELECTION
SELECT dailylabor.mason, dailylabor.foreman, dailylabor.timekeeper, dailylabor.driver, dailylabor.operator, dailylabor.labor, dailylabor.guard, dailylabor.subcon, dailylabor.date 
FROM dailylabor
LEFT JOIN dailyacco ON dailyacco.dlabor_id=dailylabor.dlabor_id
WHERE dailylabor.status='ACTIVE'
AND
dailylabor.date=(SELECT date FROM dailyacco WHERE date='?')

DAILYEQPT SELECTION
SELECT dailyeqpt.deqpt_id, dailyeqpt.eqpt_id, dailyeqpt.utilization, dailyeqpt.date
FROM dailyeqpt
LEFT JOIN dailyacco ON dailyacco.deqpt_id=dailyeqpt.deqpt_id
INNER JOIN equipment ON dailyeqpt.eqpt_id=equipment.eqpt_id
WHERE dailyeqpt.status='ACTIVE'
AND
dailyeqpt.date=(SELECT date FROM dailyacco WHERE date='?')

DAILYMAT SELECTION
SELECT dailymat.dmat_id, dailymat.material_id, dailymat.qty, dailymat.type, dailymat.date
FROM dailymat
LEFT JOIN dailyacco ON dailyacco.dmat_id=dailymat.dmat_id
INNER JOIN materials ON dailymat.material_id=materials.material_id
WHERE dailymat.status='ACTIVE'
AND
dailymat.date=(SELECT date FROM dailyacco WHERE date='?')

RECEIVED SELECTION
SELECT received.received_id, received.description, received.type, received.date
FROM received
LEFT JOIN dailyacco ON dailyacco.received_id=received.received_id
WHERE received.status='ACTIVE'
AND
received.date=(SELECT date FROM dailyacco WHERE date='?')

DAILYACCO (OTHERS) SELECTION
SELECT dailyacco.dailyacco_id, dailyacco.contract_id, dailyacco.reports_id, dailyacco.date, dailyacco.day, dailyacco.weather_am, dailyacco.weather_pm, dailyacco.meeting_conducted, dailyacco.problems, dailyacco.visitors, dailyacco.prepared_by, dailyacco.approved_by
FROM dailyacco
INNER JOIN contract ON dailyacco.contract_id=contract.contract_id
INNER JOIN reports ON dailyacco.reports_id=reports.reports_id
INNER JOIN employee ON dailyacco.prepared_by=employee.name
WHERE dailyacco.status='ACTIVE'
AND dailyacco.date='?'
